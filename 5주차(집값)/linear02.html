def ComputeJ(t0,t1,x,y):
    m=len(x)
    x_b = np.c_[np.ones((m,1)),x]
    theta = np.array([t0,t1])
    
    
    cost_vector = x_b.dot(theta) - y
    
    cost_vector = cost_vector.reshape(-1)
    
    cost2 = np.square(cost_vector)
    
    cost_sum = np.sum(cost2)/(2*m)
    
    return cost_sum
